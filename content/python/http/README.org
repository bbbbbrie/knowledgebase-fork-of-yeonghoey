  >>> from http import HTTPStatus
  >>> HTTPStatus.OK
  <HTTPStatus.OK: 200>

  True
  >>> http.HTTPStatus.OK.value
  200
  >>> HTTPStatus.OK.phrase
  'OK'
  >>> HTTPStatus.OK.description
  'Request fulfilled, document follows'

  [<HTTPStatus.CONTINUE: 100>, <HTTPStatus.SWITCHING_PROTOCOLS: 101>, ...]
