#+BEGIN_SRC html
  <!DOCTYPE html>
  <html lang="en">

  <head>
    <meta charset="utf-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>TITLE</title>
  </head>

  <body>
    <h1>HEADING</h1>
    <p>CONTENT</p>
  </body>

  </html>
#+END_SRC
