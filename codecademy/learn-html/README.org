#+BEGIN_SRC html
  <table>
    <tr> <!-- Row 1 -->
      <th></th>
      <th>Saturday</th>
      <th>Sunday</th>
    </tr>
    <tr> <!-- Row 2 -->
      <th>Morning</th>
      <td rowspan="2">Work</td>
      <td rowspan="3">Relax</td>
    </tr>
    <tr> <!-- Row 3 -->
      <th>Afternoon</th>
    </tr>
    <tr> <!-- Row 4 -->
      <th>Evening</th>
      <td>Dinner</td>
    </tr>
  </table>
#+END_SRC
