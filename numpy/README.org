  >>> a = np.ones((2, 3))
  >>> a
  array([[ 1.,  1.,  1.],
         [ 1.,  1.,  1.]])
  >>> np.sum(a, axis=0)
  array([ 2.,  2.,  2.])
  >>> np.sum(a, axis=1)
  array([ 3.,  3.])
  >>> np.sum(a, axis=1, keepdims=True)
  array([[ 3.],
         [ 3.]])
  >>> np.sum(a, axis=1).shape
  (2,)
  >>> np.sum(a, axis=1, keepdims=True).shape
  (2, 1)
