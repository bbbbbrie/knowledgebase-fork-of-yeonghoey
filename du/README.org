* du
#+BEGIN_SRC shell
  $ du
  0       ./temp
  24      .

  $ du ./temp
  0       ./temp

  # -a stands for 'all files'
  $ du -a  ./
  8       ./404.html
  8       ./CNAME
  8       ./index.html
  0       ./temp/a
  0       ./temp/b
  0       ./temp
  24      .

  # -h stands for 'human readable'
  $ du -h
  0B    ./temp
  12K    .

  # list file in order of size
  $ du -a ./ | sort -n

  # merge subdirectories
  $ du -sh
  12K    .

  # display sizes of all entries in current directory
  $ du -sh *
  4.0K    404.html
  4.0K    CNAME
  4.0K    index.html
  0B    temp
#+END_SRC
